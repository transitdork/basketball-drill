  <script>

    function selectPlayer() {

      const teams = parseInt(document.getElementById('teams').value);

      const players = parseInt(document.getElementById('players').value);


      if (isNaN(teams) || isNaN(players) || teams < 1 || teams > 4 || players < 1 || players > 5) {

        alert('Please enter valid team and player numbers.');

        return;

      }


      const team = Math.floor(Math.random() * teams) + 1;

      const player = Math.floor(Math.random() * players) + 1;

      const selection = `Team ${team} - Player ${player}`;


      document.getElementById('result').textContent = selection;


      // Speak result out loud

      const utterance = new SpeechSynthesisUtterance(selection);

      speechSynthesis.cancel();

      speechSynthesis.speak(utterance);

    }


    // Register service worker for offline use

    if ('serviceWorker' in navigator) {

      window.addEventListener('load', () => {

        navigator.serviceWorker.register('service-worker.js')

          .catch(err => console.log('Service Worker registration failed:', err));

      });

    }

  </script>
